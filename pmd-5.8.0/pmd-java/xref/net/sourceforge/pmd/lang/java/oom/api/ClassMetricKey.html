7   import net.sourceforge.pmd.lang.java.oom.metrics.AtfdMetric;
8   import net.sourceforge.pmd.lang.java.oom.metrics.CycloMetric;
9   import net.sourceforge.pmd.lang.java.oom.metrics.LocMetric;
10  import net.sourceforge.pmd.lang.java.oom.metrics.NcssMetric;
11  import net.sourceforge.pmd.lang.java.oom.metrics.WmcMetric;
12  
13  /**
14   * Keys identifying class metrics.
15   */
16  public enum ClassMetricKey implements MetricKey<ClassMetric> {
17      /** Access to Foreign Data. */
18      ATFD(new AtfdMetric()),
19      /** Weighed Method Count. */
20      WMC(new WmcMetric()),
21      /** Cyclomatic complexity. */
22      CYCLO(new CycloMetric()),
23      /** Non Commenting Source Statements. */
24      NCSS(new NcssMetric()),
25      /** Lines of Code. */
26      LOC(new LocMetric());
27  
28      private final ClassMetric calculator;
29  
30      ClassMetricKey(ClassMetric m) {
31          calculator = m;
32      }
33  
34      @Override
35      public ClassMetric getCalculator() {
36          return calculator;
37      }
38  }
