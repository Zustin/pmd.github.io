5   package net.sourceforge.pmd.lang.java.ast;
6   
7   import net.sourceforge.pmd.lang.java.typeresolution.typedefinition.JavaTypeDefinition;
8   
9   /**
10   * An extension of the SimpleJavaNode which implements the TypeNode interface.
11   *
12   * @see AbstractJavaNode
13   * @see TypeNode
14   */
15  public abstract class AbstractJavaTypeNode extends AbstractJavaNode implements TypeNode {
16      private JavaTypeDefinition typeDefinition;
17  
18      public AbstractJavaTypeNode(int i) {
19          super(i);
20      }
21  
22      public AbstractJavaTypeNode(JavaParser p, int i) {
23          super(p, i);
24      }
25  
26      @Override
27      public Class<?> getType() {
28          if (typeDefinition != null) {
29              return typeDefinition.getType();
30          }
31  
32          return null;
33      }
34  
35      @Override
36      public void setType(Class<?> type) {
37          typeDefinition = JavaTypeDefinition.build(type);
38      }
39  
40      @Override
41      public JavaTypeDefinition getTypeDefinition() {
42          return typeDefinition;
43      }
44  
45      @Override
46      public void setTypeDefinition(JavaTypeDefinition typeDefinition) {
47          this.typeDefinition = typeDefinition;
48      }
49  }
