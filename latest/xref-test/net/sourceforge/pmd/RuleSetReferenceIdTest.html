31      private static void assertRuleSetReferenceId(final boolean expectedExternal, final String expectedRuleSetFileName,
32              final boolean expectedAllRules, final String expectedRuleName, final String expectedToString,
33              final RuleSetReferenceId reference) {
34  
35          assertEquals("Wrong external", expectedExternal, reference.isExternal());
36          assertEquals("Wrong RuleSet file name", expectedRuleSetFileName, reference.getRuleSetFileName());
37          assertEquals("Wrong all Rule reference", expectedAllRules, reference.isAllRules());
38          assertEquals("Wrong Rule name", expectedRuleName, reference.getRuleName());
39          assertEquals("Wrong toString()", expectedToString, reference.toString());
40      }

83      public void testInternalWithExternalRuleSet() {
84  
85          // This is representative of how the RuleSetFactory temporarily pairs an
86          // internal reference
87          // with an external reference.
88          RuleSetReferenceId internalRuleSetReferenceId = new RuleSetReferenceId("MockRuleName");
89          assertRuleSetReferenceId(false, null, false, "MockRuleName", "MockRuleName", internalRuleSetReferenceId);
90          RuleSetReferenceId externalRuleSetReferenceId = new RuleSetReferenceId("rulesets/java/basic.xml");
91          assertRuleSetReferenceId(true, "rulesets/java/basic.xml", true, null, "rulesets/java/basic.xml",
92                  externalRuleSetReferenceId);
93  
94          RuleSetReferenceId pairRuleSetReferenceId = new RuleSetReferenceId("MockRuleName", externalRuleSetReferenceId);
95          assertRuleSetReferenceId(true, "rulesets/java/basic.xml", false, "MockRuleName",
96                  "rulesets/java/basic.xml/MockRuleName", pairRuleSetReferenceId);
97      }

111     @Test
112     public void testConstructorGivenHttpUrlInputStream() throws Exception {
113         String path = "/profiles/export?format=pmd&language=java&name=Sonar%2520way";
114         String rulesetUrl = "http://localhost:" + wireMockRule.port() + path;
115         stubFor(head(urlEqualTo(path)).willReturn(aResponse().withStatus(200)));
116         stubFor(get(urlEqualTo(path))
117                 .willReturn(aResponse().withStatus(200).withHeader("Content-type", "text/xml").withBody("xyz")));
118 
119         RuleSetReferenceId ruleSetReferenceId = new RuleSetReferenceId("  " + rulesetUrl + "  ");
120         assertRuleSetReferenceId(true, rulesetUrl, true, null, rulesetUrl, ruleSetReferenceId);
121 
122         try (InputStream inputStream = ruleSetReferenceId.getInputStream(RuleSetReferenceIdTest.class.getClassLoader());) {
123             String loaded = IOUtils.toString(inputStream, "UTF-8");
124             assertEquals("xyz", loaded);
125         }
126 
127         verify(1, headRequestedFor(urlEqualTo(path)));
128         verify(0, headRequestedFor(urlEqualTo("/profiles")));
129         verify(1, getRequestedFor(urlEqualTo(path)));
130         assertEquals(1, findAll(headRequestedFor(urlMatching(".*"))).size());
131         assertEquals(1, findAll(getRequestedFor(urlMatching(".*"))).size());
132     }

134     @Test
135     public void testConstructorGivenHttpUrlSingleRuleInputStream() throws Exception {
136         String path = "/profiles/export?format=pmd&language=java&name=Sonar%2520way";
137         String completePath = path + "/DummyBasicMockRule";
138         String hostpart = "http://localhost:" + wireMockRule.port();
139         String basicRuleSet = IOUtils
140                 .toString(RuleSetReferenceId.class.getResourceAsStream("/rulesets/dummy/basic.xml"));
141 
142         stubFor(head(urlEqualTo(completePath)).willReturn(aResponse().withStatus(404)));
143         stubFor(head(urlEqualTo(path)).willReturn(aResponse().withStatus(200).withHeader("Content-type", "text/xml")));
144         stubFor(get(urlEqualTo(path))
145                 .willReturn(aResponse().withStatus(200).withHeader("Content-type", "text/xml").withBody(basicRuleSet)));
146 
147         RuleSetReferenceId ruleSetReferenceId = new RuleSetReferenceId("  " + hostpart + completePath + "  ");
148         assertRuleSetReferenceId(true, hostpart + path, false, "DummyBasicMockRule", hostpart + completePath,
149                 ruleSetReferenceId);
150 
151         try (InputStream inputStream = ruleSetReferenceId.getInputStream(RuleSetReferenceIdTest.class.getClassLoader());) {
152             String loaded = IOUtils.toString(inputStream, "UTF-8");
153             assertEquals(basicRuleSet, loaded);
154         }
155 
156         verify(1, headRequestedFor(urlEqualTo(completePath)));
157         verify(1, headRequestedFor(urlEqualTo(path)));
158         verify(1, getRequestedFor(urlEqualTo(path)));
159         verify(0, getRequestedFor(urlEqualTo(completePath)));
160         assertEquals(2, findAll(headRequestedFor(urlMatching(".*"))).size());
161         assertEquals(1, findAll(getRequestedFor(urlMatching(".*"))).size());
162     }
