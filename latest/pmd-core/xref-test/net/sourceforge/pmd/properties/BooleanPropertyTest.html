16  public class BooleanPropertyTest extends AbstractPropertyDescriptorTester {
17  
18      public BooleanPropertyTest() {
19          super("Boolean");
20      }
21  
22      /**
23       * Method createValue.
24       *
25       * @param valueCount
26       *            int
27       * @return Object
28       */
29      @Override
30      protected Object createValue(int valueCount) {
31  
32          if (valueCount == 1) {
33              return System.currentTimeMillis() % 1 > 0 ? Boolean.TRUE : Boolean.FALSE;
34          }
35  
36          Boolean[] values = new Boolean[valueCount];
37          for (int i = 0; i < values.length; i++) {
38              values[i] = (Boolean) createValue(1);
39          }
40          return values;
41      }
42  
43      @Override
44      @Test
45      public void testErrorForBad() {
46          // override, cannot create a 'bad' boolean per se
47      }
48  
49      @Override
50      protected Object createBadValue(int count) {
51          return null;
52      }
53  
54      /**
55       * Method createProperty.
56       *
57       * @param multiValue
58       *            boolean
59       * @return PropertyDescriptor
60       */
61      @Override
62      protected PropertyDescriptor createProperty(boolean multiValue) {
63          return multiValue ? new BooleanMultiProperty("testBoolean", "Test boolean property",
64                  new Boolean[] { false, true, true }, 1.0f)
65                  : new BooleanProperty("testBoolean", "Test boolean property", false, 1.0f);
66      }
67  
68      /**
69       * Method createBadProperty.
70       *
71       * @param multiValue
72       *            boolean
73       * @return PropertyDescriptor
74       */
75      @Override
76      protected PropertyDescriptor createBadProperty(boolean multiValue) {
77          return multiValue
78                  ? new BooleanMultiProperty("", "Test boolean property", new Boolean[] { false, true, true }, 1.0f)
79                  : new BooleanProperty("testBoolean", "", false, 1.0f);
80      }
