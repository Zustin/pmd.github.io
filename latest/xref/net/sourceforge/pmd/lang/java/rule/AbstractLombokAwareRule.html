4   package net.sourceforge.pmd.lang.java.rule;
5   
6   import java.util.HashSet;
7   import java.util.List;
8   import java.util.Set;
9   
10  import net.sourceforge.pmd.lang.ast.Node;
11  import net.sourceforge.pmd.lang.java.ast.ASTAnnotation;
12  import net.sourceforge.pmd.lang.java.ast.ASTClassOrInterfaceDeclaration;
13  import net.sourceforge.pmd.lang.java.ast.ASTCompilationUnit;
14  import net.sourceforge.pmd.lang.java.ast.ASTImportDeclaration;
15  import net.sourceforge.pmd.lang.java.ast.ASTName;
16  
17  /**
18   * Base class for rules, that should ignore classes/fields that
19   * are annotated with Lombok annotations.
20   *
21   * @author Andreas Dangel
22   */
23  public class AbstractLombokAwareRule extends AbstractJavaRule {
24  
25      private boolean lombokImported = false;
26      private boolean classHasLombokAnnotation = false;
27      private static final String LOMBOK_PACKAGE = "lombok";
28      private static final Set<String> LOMBOK_ANNOTATIONS = new HashSet<>();
29      static {
30          LOMBOK_ANNOTATIONS.add("Data");
31          LOMBOK_ANNOTATIONS.add("Getter");
32          LOMBOK_ANNOTATIONS.add("Setter");
33          LOMBOK_ANNOTATIONS.add("Value");
34          LOMBOK_ANNOTATIONS.add("RequiredArgsConstructor");
35          LOMBOK_ANNOTATIONS.add("AllArgsConstructor");
36          LOMBOK_ANNOTATIONS.add("Builder");
37      }
38  
39      @Override
40      public Object visit(ASTCompilationUnit node, Object data) {
41          lombokImported = false;
42          return super.visit(node, data);
43      }
44  
45      @Override
46      public Object visit(ASTImportDeclaration node, Object data) {
47          ASTName name = node.getFirstChildOfType(ASTName.class);
48          if (!lombokImported && name != null && name.getImage() != null & name.getImage().startsWith(LOMBOK_PACKAGE)) {
49              lombokImported = true;
50          }
51          return super.visit(node, data);
52      }
53  
54      @Override
55      public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
56          classHasLombokAnnotation = hasLombokAnnotation(node);
57          return super.visit(node, data);
58      }
59  
60      /**
61       * Returns whether there have been class level Lombok annotations found.
62       * Note: this can only be queried after the class declaration node has been processed.
63       * @return <code>true</code> if a lombok annotation at the class level has been found
64       */
65      protected boolean hasClassLombokAnnotation() {
66          return classHasLombokAnnotation;
67      }
68  
69      /**
70       * Checks whether the given node is annotated with any lombok annotation.
71       * The node can be any node, e.g. class declaration or field declaration.
72       * @param node the node to check
73       * @return <code>true</code> if a lombok annotation has been found
74       */
75      protected boolean hasLombokAnnotation(Node node) {
76          boolean result = false;
77          Node parent = node.jjtGetParent();
78          List<ASTAnnotation> annotations = parent.findChildrenOfType(ASTAnnotation.class);
79          for (ASTAnnotation annotation : annotations) {
80              ASTName name = annotation.getFirstDescendantOfType(ASTName.class);
81              if (name != null) {
82                  String annotationName = name.getImage();
83                  if (lombokImported) {
84                      if (LOMBOK_ANNOTATIONS.contains(annotationName)) {
85                          result = true;
86                      }
87                  } else {
88                      if (annotationName.startsWith(LOMBOK_PACKAGE + ".")) {
89                          String shortName = annotationName.substring(LOMBOK_PACKAGE.length() + 1);
90                          if (LOMBOK_ANNOTATIONS.contains(shortName)) {
91                              result = true;
92                          }
93                      }
94                  }
95              }
96          }
97          return result;
98      }
99  }
