38  /**
39   * Visitor for scope creation. Visits all nodes of an AST and creates scope
40   * objects for nodes representing syntactic entities which may contain
41   * declarations. For example, a block may contain variable definitions (which
42   * are declarations) and therefore needs a scope object where these declarations
43   * can be associated, whereas an expression can't contain declarations and
44   * therefore doesn't need a scope object. With the exception of global scopes,
45   * each scope object is linked to its parent scope, which is the scope object of
46   * the next embedding syntactic entity that has a scope.
47   */
48  public class ScopeAndDeclarationFinder extends JavaParserVisitorAdapter {
49  
50      private ClassLoader classLoader;
51  
52      /**
53       * A stack of scopes reflecting the scope hierarchy when a node is visited.
54       * This is used to set the parents of the created scopes correctly.
55       */
56      private Deque<Scope> scopes = new ArrayDeque<>();
57  
58      /**
59       * Creates a new {@link ScopeAndDeclarationFinder} using the current class
60       * loader.
61       */
62      public ScopeAndDeclarationFinder() {
63          this(ScopeAndDeclarationFinder.class.getClassLoader());
64      }
65  
66      /**
67       * Creates a new {@link ScopeAndDeclarationFinder}.
68       * 
69       * @param classLoader
70       *            the class loader to use to resolve types, see
71       *            {@link SourceFileScope} and {@link TypeSet}
72       */
73      public ScopeAndDeclarationFinder(ClassLoader classLoader) {
74          this.classLoader = classLoader;
75      }
76  
77      /**
78       * Sets the scope of a node and adjusts the scope stack accordingly. The
79       * scope on top of the stack is set as the parent of the given scope, which
80       * is then also stored on the scope stack.
81       *
82       * @param newScope
83       *            the scope for the node.
84       * @param node
85       *            the AST node for which the scope is to be set.
86       * @throws java.util.EmptyStackException
87       *             if the scope stack is empty.
88       */
89      private void addScope(Scope newScope, JavaNode node) {
90          newScope.setParent(scopes.peek());
91          scopes.push(newScope);
92          node.setScope(newScope);
93      }
94  
95      /**
96       * Creates a new local scope for an AST node. The scope on top of the stack
97       * is set as the parent of the new scope, which is then also stored on the
98       * scope stack.
99       *
100      * @param node
101      *            the AST node for which the scope has to be created.
102      * @throws java.util.EmptyStackException
103      *             if the scope stack is empty.
104      */
105     private void createLocalScope(JavaNode node) {
106         addScope(new LocalScope(), node);
107     }
108 
109     /**
110      * Creates a new method scope for an AST node. The scope on top of the stack
111      * is set as the parent of the new scope, which is then also stored on the
112      * scope stack.
113      *
114      * @param node
115      *            the AST node for which the scope has to be created.
116      * @throws java.util.EmptyStackException
117      *             if the scope stack is empty.
118      */
119     private void createMethodScope(JavaNode node) {
120         addScope(new MethodScope(node), node);
121     }
122 
123     /**
124      * Creates a new class scope for an AST node. The scope on top of the stack
125      * is set as the parent of the new scope, which is then also stored on the
126      * scope stack.
127      *
128      * @param node
129      *            the AST node for which the scope has to be created.
130      * @throws java.util.EmptyStackException
131      *             if the scope stack is empty.
132      */
133     private void createClassScope(JavaNode node) {
134         Scope s = ((JavaNode) node.jjtGetParent()).getScope();
135         ClassNameDeclaration classNameDeclaration = new ClassNameDeclaration(node);
136         s.addDeclaration(classNameDeclaration);
137 
138         if (node instanceof ASTClassOrInterfaceBody) {
139             addScope(new ClassScope(classNameDeclaration), node);
140         } else {
141             addScope(new ClassScope(node.getImage(), classNameDeclaration), node);
142         }
143     }
144 
145     /**
146      * Creates a new global scope for an AST node. The new scope is stored on
147      * the scope stack.
148      *
149      * @param node
150      *            the AST node for which the scope has to be created.
151      */
152     private void createSourceFileScope(ASTCompilationUnit node) {
153         // When we do full symbol resolution, we'll need to add a truly
154         // top-level GlobalScope.
155         SourceFileScope scope;
156         ASTPackageDeclaration n = node.getPackageDeclaration();
157         if (n != null) {
158             scope = new SourceFileScope(classLoader, n.jjtGetChild(0).getImage());
159         } else {
160             scope = new SourceFileScope(classLoader);
161         }
162         scope.configureImports(node.findChildrenOfType(ASTImportDeclaration.class));
163 
164         scopes.push(scope);
165         node.setScope(scope);
166     }
167 
168     @Override
169     public Object visit(ASTCompilationUnit node, Object data) {
170         createSourceFileScope(node);
171         cont(node);
172         return data;
173     }
174 
175     @Override
176     public Object visit(ASTClassOrInterfaceDeclaration node, Object data) {
177         createClassScope(node);
178         cont(node);
179         return data;
180     }

182     @Override
183     public Object visit(ASTEnumDeclaration node, Object data) {
184         createClassScope(node);
185         ((ClassScope) node.getScope()).setIsEnum(true);
186         cont(node);
187         return data;
188     }

197     @Override
198     public Object visit(ASTClassOrInterfaceBody node, Object data) {
199         if (node.isAnonymousInnerClass() || node.isEnumChild()) {
200             createClassScope(node);
201             cont(node);
202         } else {
203             super.visit(node, data);
204         }
205         return data;
206     }

229     @Override
230     public Object visit(ASTConstructorDeclaration node, Object data) {
231         /*
232          * Local variables declared inside the constructor need to be in a
233          * different scope so special handling is needed
234          */
235         createMethodScope(node);
236 
237         Scope methodScope = node.getScope();
238 
239         Node formalParameters = node.jjtGetChild(0);
240         int i = 1;
241         int n = node.jjtGetNumChildren();
242         if (!(formalParameters instanceof ASTFormalParameters)) {
243             visit((ASTTypeParameters) formalParameters, data);
244             formalParameters = node.jjtGetChild(1);
245             i++;
246         }
247         visit((ASTFormalParameters) formalParameters, data);
248 
249         Scope localScope = null;
250         for (; i < n; i++) {
251             JavaNode b = (JavaNode) node.jjtGetChild(i);
252             if (b instanceof ASTBlockStatement) {
253                 if (localScope == null) {
254                     createLocalScope(node);
255                     localScope = node.getScope();
256                 }
257                 b.setScope(localScope);
258                 visit(b, data);
259             } else {
260                 visit(b, data);
261             }
262         }
263         if (localScope != null) {
264             // pop the local scope
265             scopes.pop();
266 
267             // reset the correct scope for the constructor
268             node.setScope(methodScope);
269         }
270         // pop the method scope
271         scopes.pop();
272 
273         return data;
274     }
275 
276     @Override
277     public Object visit(ASTMethodDeclaration node, Object data) {
278         createMethodScope(node);
279         ASTMethodDeclarator md = node.getFirstChildOfType(ASTMethodDeclarator.class);
280         node.getScope().getEnclosingScope(ClassScope.class).addDeclaration(new MethodNameDeclaration(md));
281         cont(node);
282         return data;
283     }

314     @Override
315     public Object visit(ASTVariableDeclaratorId node, Object data) {
316         VariableNameDeclaration decl = new VariableNameDeclaration(node);
317         node.getScope().addDeclaration(decl);
318         node.setNameDeclaration(decl);
319         return super.visit(node, data);
320     }
321 
322     @Override
323     public Object visit(ASTSwitchStatement node, Object data) {
324         createLocalScope(node);
325         cont(node);
326         return data;
327     }
328 
329     private void cont(AbstractJavaNode node) {
330         super.visit(node, null);
331         scopes.pop();
332     }
