7   /**
8    * These are the possible Rule priority values.
9    *
10   * For backward compatibility, priorities range in value from 1 to 5, with 5
11   * being the lowest priority. This means the ordinal value of the Enum should be
12   * avoided in favor of {@link RulePriority#getPriority()} and
13   * {@link RulePriority#valueOf(int)}
14   *
15   * @see <a href="http://pmd.sourceforge.net/pmd-5.1.2/rule-guidelines.html">How
16   *      to define rules priority</a>
17   */
18  public enum RulePriority {
19  
20      /** High: Change absolutely required. Behavior is critically broken/buggy */
21      HIGH(1, "High"),
22      /**
23       * Medium to high: Change highly recommended. Behavior is quite likely to be
24       * broken/buggy.
25       */
26      MEDIUM_HIGH(2, "Medium High"),
27      /**
28       * Medium: Change recommended. Behavior is confusing, perhaps buggy, and/or
29       * against standards/best practices.
30       */
31      MEDIUM(3, "Medium"),
32      /**
33       * Medium to low: Change optional. Behavior is not likely to be buggy, but
34       * more just flies in the face of standards/style/good taste.
35       */
36      MEDIUM_LOW(4, "Medium Low"),
37      /**
38       * Low: Change highly optional. Nice to have, such as a consistent naming
39       * policy for package/class/fields...
40       */
41      LOW(5, "Low");
42  
43      private final int priority;
44      private final String name;
45  
46      RulePriority(int priority, String name) {
47          this.priority = priority;
48          this.name = name;
49      }
50  
51      /**
52       * Get the priority value as a number. This is the value to be used in the
53       * externalized form of a priority (e.g. in RuleSet XML).
54       *
55       * @return The <code>int</code> value of the priority.
56       */
57      public int getPriority() {
58          return priority;
59      }
60  
61      /**
62       * Get the descriptive name of this priority.
63       *
64       * @return The descriptive name.
65       */
66      public String getName() {
67          return name;
68      }
69  
70      /**
71       * Returns the descriptive name of the priority.
72       *
73       * @return descriptive name of the priority
74       * @see #getName()
75       */
76      @Override
77      public String toString() {
78          return name;
79      }
80  
81      /**
82       * Get the priority which corresponds to the given number as returned by
83       * {@link RulePriority#getPriority()}. If the number is an invalid value,
84       * then {@link RulePriority#LOW} will be returned.
85       *
86       * @param priority
87       *            The numeric priority value.
88       * @return The priority.
89       */
90      public static RulePriority valueOf(int priority) {
91          try {
92              return RulePriority.values()[priority - 1];
93          } catch (ArrayIndexOutOfBoundsException e) {
94              return LOW;
95          }
96      }
97  }
