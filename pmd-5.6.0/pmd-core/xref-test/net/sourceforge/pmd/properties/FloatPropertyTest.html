5   package net.sourceforge.pmd.properties;
6   
7   import net.sourceforge.pmd.PropertyDescriptor;
8   import net.sourceforge.pmd.lang.rule.properties.FloatMultiProperty;
9   import net.sourceforge.pmd.lang.rule.properties.FloatProperty;
10  
11  /**
12   * Evaluates the functionality of the FloatProperty descriptor by testing its
13   * ability to catch creation errors (illegal args), flag out-of-range test
14   * values, and serialize/deserialize groups of float values onto/from a string
15   * buffer.
16   *
17   * @author Brian Remedios
18   */
19  public class FloatPropertyTest extends AbstractPropertyDescriptorTester {
20  
21      private static final float MIN = 1.0f;
22      private static final float MAX = 11.0f;
23      private static final float SHIFT = 3.0f;
24  
25      public static FloatProperty randomProperty(int nameLength, int descLength, boolean multiValue) {
26  
27          float defalt = randomFloat(0, 1000f);
28  
29          return new FloatProperty(randomString(nameLength), randomString(descLength), defalt - 1000f, defalt + 1000,
30                  defalt, 0f);
31      }
32  
33      public FloatPropertyTest() {
34          super("Float");
35      }
36  
37      /**
38       * Method createValue.
39       *
40       * @param count
41       *            int
42       * @return Object
43       */
44      @Override
45      protected Object createValue(int count) {
46  
47          if (count == 1) {
48              return Float.valueOf(randomFloat(MIN, MAX));
49          }
50  
51          Float[] values = new Float[count];
52          for (int i = 0; i < values.length; i++) {
53              values[i] = (Float) createValue(1);
54          }
55          return values;
56      }
57  
58      /**
59       * Creates and returns (count) number of out-of-range float values
60       *
61       * @param count
62       *            int
63       * @return Object
64       */
65      @Override
66      protected Object createBadValue(int count) {
67  
68          if (count == 1) {
69              return Float.valueOf(randomBool() ? randomFloat(MIN - SHIFT, MIN) : randomFloat(MAX, MAX + SHIFT));
70          }
71  
72          Float[] values = new Float[count];
73          for (int i = 0; i < values.length; i++) {
74              values[i] = (Float) createBadValue(1);
75          }
76          return values;
77      }
78  
79      /**
80       * Method createProperty.
81       *
82       * @param multiValue
83       *            boolean
84       * @return PropertyDescriptor
85       */
86      @Override
87      protected PropertyDescriptor createProperty(boolean multiValue) {
88  
89          return multiValue ? new FloatMultiProperty("testFloat", "Test float property", MIN, MAX,
90                  new Float[] { -1f, 0f, 1f, 2f }, 1.0f)
91                  : new FloatProperty("testFloat", "Test float property", MIN, MAX, 9.0f, 1.0f);
92      }
93  
94      /**
95       * Method createBadProperty.
96       *
97       * @param multiValue
98       *            boolean
99       * @return PropertyDescriptor
100      */
101     @Override
102     protected PropertyDescriptor createBadProperty(boolean multiValue) {
103 
104         return multiValue ? new FloatMultiProperty("testFloat", "Test float property", 0f, 5f,
105                 new Float[] { -1f, 0f, 1f, 2f }, 1.0f)
106                 : new FloatProperty("testFloat", "Test float property", 5f, 4f, 9.0f, 1.0f);
107     }
108 }
